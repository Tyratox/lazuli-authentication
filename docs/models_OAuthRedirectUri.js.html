/**
  * Generates the oauth redirect uri sequelize model
  * @param {Object} sequelize The sequelize object to define the model on
  * @param {Object} DataTypes Sequelize datatypes
  */
const OAuthRedirectUri = (sequelize, DataTypes) => {
	let OAuthRedirectUri = sequelize.define(
		"oauth_redirect_uri",
		{
			uri: {
				type: Sequelize.STRING
			}
		},
		{
			charset: "utf8",
			collate: "utf8_unicode_ci"
		}
	);

	/**
	 * Associates this model with others
	 * @param  {Object} models An object containing all registered database models
	 * @return {void}
	 */
	OAuthRedirectUri.associate = function({ OAuthClient }) {
		this.belongsTo(OAuthClient, {
			as: "OAuthClient",
			foreignKey: "oauth_client_id"
		});
	};

	/**
	 * Converts a model to a json object
	 * @param  {Object} user The user accessing this information
	 * @return {Object} The generated json object
	 */
	OAuthRedirectUri.toJSON = function(user) {
		return { id: this.get("id"), uri: this.get("uri") };
	};

	return OAuthRedirectUri;
};

module.exports = OAuthRedirectUri;
