/**
 * Generates the oauth provider sequelize model
 * @param {Object} sequelize The sequelize object to define the model on
 * @param {Object} DataTypes Sequelize datatypes
 */
const OAuthProvider = (sequelize, DataTypes) => {
	let OAuthProvider = sequelize.define(
		"oauth_provider",
		{
			type: {
				type: DataTypes.ENUM,
				values: ["google", "facebook"]
			},
			accessToken: {
				type: DataTypes.STRING
			},
			refreshToken: {
				type: DataTypes.STRING
			}
		},
		{
			charset: "utf8",
			collate: "utf8_unicode_ci"
		}
	);

	/**
	 * Associates this model with others
	 * @param  {Object} models An object containing all registered database models
	 * @return {void}
	 */
	OAuthProvider.associate = function({ User }) {
		this.belongsTo(User, {
			as: "User",
			foreignKey: "user_id"
		});
	};

	/**
	 * Converts a model to a json object
	 * @param  {Object} user The user accessing this information
	 * @return {Object} The generated json object
	 */
	OAuthProvider.prototype.toJSON = function(user) {
		let provider = this.get();

		let json = pick(provider, ["id", "type", "createdAt", "updatedAt"]);

		if (
			user &&
			(provider.user_id === user.get("id") ||
				user.doesHavePermission("admin.oauthProvider.readOthers"))
		) {
			json.userId = provider.user_id;

			if (provider.User) {
				json.user = provider.User.toJSON(options);
			}
		}

		if (options.admin) {
			json.accessToken = provider.accessToken;
			json.refreshToken = provider.refreshToken;
		}

		return json;
	};

	return OAuthProvider;
};

module.exports = OAuthProvider;
